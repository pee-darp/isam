203 bool SparseVector::set(int idx, const VectorXd& vals) {
204  bool created_entry = false;
205  int n = 0;
206  int c = vals.rows();
207 
208  // First check if we can append
209  if (_nnz > 0 && idx > _indices[_nnz-1]) {
210  n = _nnz;
211  } else {
212  n = _search(idx); // search closest entry
213  }
214 
215  if (n<_nnz && _indices[n] == idx) {
216  // entry exists? then overwrite!
217  // BIG ASSUMPTION when writing multiple values:
218  // they either all exist or they don't
219  for (int i=0;i<c;i++) {
220  _values[n+i] = vals(i);
221  }
222  } else {
223  // new entries have to be created
224  created_entry = true;
225  // enough space left?
226  if (_nnz+c > _nnz_max) {
227  // automatic resizing, amortized cost O(1)
228  int new_nnz_max = _nnz_max*2 + c;
229  _resize(new_nnz_max);
230  }
231  // need to make space in the middle?
232  if (n<_nnz) {
233  memmove(&(_indices[n+c]), &(_indices[n]), (_nnz-n)*sizeof(int));
234  memmove(&(_values[n+c]), &(_values[n]), (_nnz-n)*sizeof(double));
235  }
236  // insert new entry
237  _nnz+=c;
238  for (int i=0;i<c;i++) {
239  _indices[n+i] = idx+i;
240  _values[n+i] = vals(i);
241  }
242  }
243 
244  return created_entry;
245 }

253  if (n!=_nnz-1) {
