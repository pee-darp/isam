301 void SparseMatrix::apply_givens(int row, int col, double* c_givens, double* s_givens) {
302  requireDebug(row>=0 && row<_num_rows && col>=0 && col<_num_cols, "SparseMatrix::apply_givens: index outside matrix.");
303  requireDebug(row>col, "SparseMatrix::apply_givens: can only zero entries below the diagonal.");
304  const SparseVector& row_top = *_rows[col];
305  const SparseVector& row_bot = *_rows[row];
306  double a = row_top(col);
307  double b = row_bot(col);
308  double c, s;
309  givens(a, b, c, s);
310  if (c_givens) *c_givens = c;
311  if (s_givens) *s_givens = s;
312 
313  int n = row_bot.nnz() + row_top.nnz();
314 
315  SparseVector_p new_row_top = new SparseVector(n);
316  SparseVector_p new_row_bot = new SparseVector(n);
317  SparseVectorIter iter_top(row_top);
318  SparseVectorIter iter_bot(row_bot);
319  bool top_valid = iter_top.valid();
320  bool bot_valid = iter_bot.valid();
321  while (top_valid || bot_valid) {
322  double val_top = 0.;
323  double val_bot = 0.;
324  int idx_top = (top_valid)?iter_top.get(val_top):-1;
325  int idx_bot = (bot_valid)?iter_bot.get(val_bot):-1;
326  int idx;
327  if (idx_bot<0) {
328  idx = idx_top;
329  } else if (idx_top<0) {
330  idx = idx_bot;
331  } else {
332  idx = min(idx_top, idx_bot);
333  }
334  if (top_valid) {
335  if (idx==idx_top) {
336  iter_top.next();
337  } else {
338  val_top = 0.;
339  }
340  }
341  if (bot_valid) {
342  if (idx==idx_bot) {
343  iter_bot.next();
344  } else {
345  val_bot = 0.;
346  }
347  }
348  double new_val_top = c*val_top - s*val_bot;
349  double new_val_bot = s*val_top + c*val_bot;
350  // remove numerically zero values to keep sparsity
351  if (fabs(new_val_top) >= NUMERICAL_ZERO) {
352  // append for O(1) operation - even O(log n) is too
353  // slow here, because this is called extremely often!
354  new_row_top->append(idx, new_val_top);
355  }
356  if (fabs(new_val_bot) >= NUMERICAL_ZERO) {
357  new_row_bot->append(idx, new_val_bot);
358  }
359  top_valid = iter_top.valid();
360  bot_valid = iter_bot.valid();
361  }
362 
363  delete _rows[col];
364  delete _rows[row];
365 
366  _rows[col] = new_row_top;
367  _rows[row] = new_row_bot;
368  _rows[row]->remove(col); // by definition, this entry is exactly 0
369 }
