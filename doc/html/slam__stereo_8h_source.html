94  StereoMeasurement project(const Pose3d& pose, const Point3dh& Xw) const {
95  Point3dh X = pose.transform_to(Xw);
96  // camera system has z pointing forward, instead of x
97  double x = X.y();
98  double y = X.z();
99  double z = X.x();
100  double w = X.w();
101  // left camera
102  double fz = _f / z;
103  double u = x * fz + _pp(0);
104  double v = y * fz + _pp(1);
105  // right camera
106  double u2 = (x - w*_b) * fz + _pp(0);
107  bool valid = ((w==0&&z>0) || (z/w) > 0.); // infront of camera?
108 
109  // Setting a point valid can make the system (hj 25/10/2012)
110  // ill conditioned, i.e. if point goes behind all cameras that see it
111  //valid = true;
112 
113  return StereoMeasurement(u, v, u2, valid);
114  }

117  Point3dh backproject(const Pose3d& pose, const StereoMeasurement& measure) const {
118  double lx = (measure.u-_pp(0))*_b;
119  double ly = (measure.v-_pp(1))*_b;
120  double lz = _f*_b;
121  double lw = measure.u - measure.u2;
122  if (lw<0.) {
123  std::cout << "Warning: StereoCamera.backproject called with negative disparity\n";
124  }
125  Point3dh X(lz, lx, ly, lw);
126 
127  return pose.transform_from(X);
