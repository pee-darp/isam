65 VectorXd Optimizer::compute_dog_leg(double alpha, const VectorXd& h_sd,
66  const VectorXd& h_gn, double delta, double& gain_ratio_denominator) {
67  if (h_gn.norm() <= delta) {
68  gain_ratio_denominator = current_SSE_at_linpoint;
69  return h_gn;
70  }
71 
72  double h_sd_norm = h_sd.norm();
73 
74  if ((alpha * h_sd_norm) >= delta) {
75  gain_ratio_denominator = delta * (2 * alpha * h_sd_norm - delta)
76  / (2 * alpha);
77  return (delta / h_sd_norm) * h_sd;
78  } else {
79  // complicated case: calculate intersection of trust region with
80  // line between Gauss-Newton and steepest descent solutions
81  VectorXd a = alpha * h_sd;
82  VectorXd b = h_gn;
83  double c = a.dot(b - a);
84  double b_a_norm2 = (b - a).squaredNorm();
85  double a_norm2 = a.squaredNorm();
86  double delta2 = delta * delta;
87  double sqrt_term = sqrt(c * c + b_a_norm2 * (delta2 - a_norm2));
88  double beta;
89  if (c <= 0) {
90  beta = (-c + sqrt_term) / b_a_norm2;
91  } else {
92  beta = (delta2 - a_norm2) / (c + sqrt_term);
93  }
94 
95  gain_ratio_denominator = .5 * alpha * (1 - beta) * (1 - beta) * h_sd_norm
96  * h_sd_norm + beta * (2 - beta) * current_SSE_at_linpoint;
97  return (alpha * h_sd + beta * (h_gn - alpha * h_sd));
98  }
99 }

110 void Optimizer::relinearize(const Properties& prop) {
111  // We're going to relinearize about the current estimate.
112  function_system.estimate_to_linpoint();
113 
114  // prepare factorization
115  SparseSystem jac = function_system.jacobian();
116 
117  // factorization and new rhs based on new linearization point will be in _R
118  VectorXd h_gn = compute_gauss_newton_step(jac, &function_system._R); // modifies _R
119 
120  if (prop.method == DOG_LEG) {
121  // Compute the gradient and cache it.
122  gradient = mul_SparseMatrixTrans_Vector(jac, jac.rhs());
123 
124  //Get the value of the sum-of-squared errors at the current linearization point.
125  current_SSE_at_linpoint = jac.rhs().squaredNorm();
126 
127  // NB: alpha's denominator will be zero iff the gradient vector is zero
128  // (since J is full-rank by hypothesis). But the gradient is zero iff
129  // we're already at the minimum, so we don't actually need to to any
130  // updates; just set the estimate to be the current linearization point,
131  // since we're already at the minimum.
132 
133  double alpha_denominator = (jac * gradient).squaredNorm();
134 
135  if (alpha_denominator > 0) {
136  double alpha_numerator = gradient.squaredNorm();
137  double alpha = alpha_numerator / alpha_denominator;
138 
139  // These values will be used to update the estimate of Delta
140  double F_0, F_h;
141 
142  F_0 = current_SSE_at_linpoint;
143 
144  double rho_denominator, rho;
145 
146  VectorXd h_dl;
147 
148  do {
149  // We repeat the computation of the dog-leg step, shrinking the
150  // trust-region radius if necessary, until we generate a sufficiently
151  // small region of trust that we accept the proposed step.
152 
153  // Compute dog-leg step.
154  // NOTE: Here we use -h_gn because of the weird sign change in the exmap functions.
155  h_dl = compute_dog_leg(alpha, -gradient, -h_gn, Delta, rho_denominator);
156 
157  // Update the estimate.
158  // NOTE: Here we use -h_dl because of the weird sign change in the exmap functions.
159  function_system.apply_exmap(-h_dl);
160 
161  // Get the value of the sum-of-squared errors at the new estimate.
162  F_h = function_system.weighted_errors(ESTIMATE).squaredNorm();
163 
164  // Compute gain ratio.
165  rho = (F_0 - F_h) / (rho_denominator);
166 
167  update_trust_radius(rho, h_dl.norm());
168  } while (rho < 0);
169 
170  // Cache last accepted dog-leg step
171  last_accepted_hdl = h_dl;
172  } else {
173  function_system.linpoint_to_estimate();
174  last_accepted_hdl = VectorXd::Zero(gradient.size());
175  }
176 
177  } else {
178  // For Gauss-Newton just apply the update directly.
179  function_system.apply_exmap(h_gn);
180  }
181 }

192 void Optimizer::augment_sparse_linear_system(SparseSystem& W,
193  const Properties& prop) {
194  if (prop.method == DOG_LEG) {
195  // We're using the incremental version of Powell's Dog-Leg, so we need
196  // to form the updated gradient.
197  const VectorXd& f_new = W.rhs();
198 
199  // Augment the running count for the sum-of-squared errors at the current
200  // linearization point.
201  current_SSE_at_linpoint += f_new.squaredNorm();
202 
203  // Allocate the new gradient vector
204  VectorXd g_new(W.num_cols());
205 
206  // Compute W^T \cdot f_new
207  VectorXd increment = mul_SparseMatrixTrans_Vector(W, f_new);
208 
209  // Set g_new = (g_old 0)^T + W^T f_new.
210  g_new.head(gradient.size()) = gradient + increment.head(gradient.size());
211  g_new.tail(W.num_cols() - gradient.size()) = increment.tail(
212  W.num_cols() - gradient.size());
213 
214  // Cache the new gradient vector
215  gradient = g_new;
216  }
217 
218  // Apply Givens to QR factorize the newly augmented sparse system.
219  for (int i = 0; i < W.num_rows(); i++) {
220  SparseVector new_row = W.get_row(i);
221  function_system._R.add_row_givens(new_row, W.rhs()(i));
222  }

225 void Optimizer::update_estimate(const Properties& prop) {
226  // Solve for the Gauss-Newton step.
227  VectorXd h_gn_reordered = function_system._R.solve();
228 
229  // permute from R-ordering to J-ordering
230  VectorXd h_gn(h_gn_reordered.size());
231  permute_vector(h_gn_reordered, h_gn, function_system._R.r_to_a());
232 
233  if (prop.method == GAUSS_NEWTON) {
234  function_system.apply_exmap(h_gn);
235  } else { //method == DOG_LEG
236  // Compute alpha. Note that since the variable ordering of the factor
237  // R differs from that of the original Jacobian J,
238  // we must first rearrange the ordering of the elements in the gradient.
239  VectorXd reordered_gradient(gradient.size());
240 
241  // Permute from J-ordering to R-ordering
242  permute_vector(gradient, reordered_gradient, function_system._R.a_to_r());
243 
244  double alpha_denominator =
245  (function_system._R * reordered_gradient).squaredNorm();
246 
247  if (alpha_denominator > 0) {
248  double alpha = (gradient.squaredNorm()) / alpha_denominator;
249 
250  double rho_denominator;
251 
252  VectorXd h_dl = compute_dog_leg(alpha, -gradient, -h_gn, Delta,
253  rho_denominator);
254  function_system.apply_exmap(-h_dl);
255 
256  // Compute the gain ratio
257  double rho = (current_SSE_at_linpoint
258  - function_system.weighted_errors(ESTIMATE).squaredNorm())
259  / rho_denominator;
260 
261  update_trust_radius(rho, h_dl.norm());
262 
263  if (rho < 0) {
264  // The proposed update actually /increased/ the value of the
265  // objective function; restore the last good estimate that we had
266  VectorXd restore_step(gradient.size());
267  restore_step.head(last_accepted_hdl.size()) = last_accepted_hdl;
268  restore_step.tail(gradient.size() - last_accepted_hdl.size()).setZero();
269 
270  function_system.apply_exmap(-restore_step);
271  } else {
272  // The proposed update was accepted; cache the dog-leg step used
273  // to produce it.
274  last_accepted_hdl = h_dl;
275  }
276  }
277 
278  // NOTE: The negatives prepended to "compute_dog_leg()" and "h_gn"
279  // are due to the weird sign change in the exmap functions.
280  }

283 void Optimizer::gauss_newton(const Properties& prop, int* num_iterations) {
284  // Batch optimization
285  int num_iter = 0;
286 
287  // Set the new linearization point to be the current estimate.
288  function_system.estimate_to_linpoint();
289  // Compute Jacobian about current estimate.
290  SparseSystem jacobian = function_system.jacobian();
291 
292  // Get the current error residual vector
293  VectorXd r = function_system.weighted_errors(LINPOINT);
294 
295 #ifdef USE_PDL_STOPPING_CRITERIA
296  // Compute the current gradient direction vector
297  VectorXd g = mul_SparseMatrixTrans_Vector(jacobian, r);
298 #else
299  double error = r.squaredNorm();
300  double error_new;
301  // We haven't computed a step yet, so this initialization is to ensure
302  // that we never skip over the while loop as a result of failing
303  // change-in-error check.
304  double error_diff = prop.epsilon_rel * error + 1;
305 #endif
306 
307  // Compute Gauss-Newton step h_{gn} to get to the next estimated optimizing point.
308  VectorXd delta = compute_gauss_newton_step(jacobian);
309 
310  while (
311  // We ALWAYS use these criteria
312  ((prop.max_iterations <= 0) || (num_iter < prop.max_iterations))
313  && (delta.norm() > prop.epsilon2)
314 
315 #ifdef USE_PDL_STOPPING_CRITERIA
316  && (r.lpNorm<Eigen::Infinity>() > prop.epsilon3)
317  && (g.lpNorm<Eigen::Infinity>() > prop.epsilon1)
318 
319 #else // Custom stopping criteria for GN
320  && (error > prop.epsilon_abs)
321  && (fabs(error_diff) > prop.epsilon_rel * error)
322 #endif
323 
324  ) // end while conditional
325  {
326  num_iter++;
327 
328  // Apply the Gauss-Newton step h_{gn} to...
329  function_system.apply_exmap(delta);
330  // ...set the new linearization point to be the current estimate.
331  function_system.estimate_to_linpoint();
332  // Relinearize about the new current estimate.
333  jacobian = function_system.jacobian();
334  // Compute the error residual vector at the new estimate.
335  r = function_system.weighted_errors(LINPOINT);
336 
337 #ifdef USE_PDL_STOPPING_CRITERIA
338  g = mul_SparseMatrixTrans_Vector(jacobian, r);
339 #else
340  // Update the error difference in errors between the previous and
341  // current estimates.
342  error_new = r.squaredNorm();
343  error_diff = error - error_new;
344  error = error_new; // Record the absolute error at the current estimate
345 #endif
346 
347  // Compute Gauss-Newton step h_{gn} to get to the next estimated
348  // optimizing point.
349  delta = compute_gauss_newton_step(jacobian);
350  if (!prop.quiet) {
351  cout << "Iteration " << num_iter << ": residual ";
352 
353 #ifdef USE_PDL_STOPPING_CRITERIA
354  cout << r.squaredNorm();
355 #else
356  cout << error;
357 #endif
358  cout << endl;
359  }
360  } //end while
361 
362  if (num_iterations != NULL) {
363  *num_iterations = num_iter;
364  }
365  _cholesky->get_R(function_system._R);
366 }

368 void Optimizer::levenberg_marquardt(const Properties& prop,
369  int* num_iterations) {
370  int num_iter = 0;
371  double lambda = prop.lm_lambda0;
372  // Using linpoint as current estimate below.
373  function_system.estimate_to_linpoint();
374 
375  // Get the current Jacobian at the linearization point.
376  SparseSystem jacobian = function_system.jacobian();
377 
378  // Get the error residual vector at the current linearization point.
379  VectorXd r = function_system.weighted_errors(LINPOINT);
380 
381  // Record the absolute sum-of-squares error (i.e., objective function value) here.
382  double error = r.squaredNorm();
383 
384 #ifdef USE_PDL_STOPPING_CRITERIA
385  // Compute the gradient direction vector at the current linearization point
386  VectorXd g = mul_SparseMatrixTrans_Vector(jacobian, r);
387 #endif
388 
389  double error_diff, error_new;
390 
391  // solve at J'J + lambda*diag(J'J)
392  VectorXd delta = compute_gauss_newton_step(jacobian, &function_system._R,
393  lambda);
394 
395  while (
396  // We ALWAYS use these stopping criteria
397  ((prop.max_iterations <= 0) || (num_iter < prop.max_iterations))
398  && (delta.norm() > prop.epsilon2)
399 
400 #ifdef USE_PDL_STOPPING_CRITERIA
401  && (r.lpNorm<Eigen::Infinity>() > prop.epsilon3)
402  && (g.lpNorm<Eigen::Infinity>() > prop.epsilon1)
403 #else
404  && (error > prop.epsilon_abs)
405 #endif
406  ) // end while conditional
407  {
408  num_iter++;
409 
410  // remember the last accepted linearization point
411  function_system.linpoint_to_estimate();
412  // Apply the delta vector DIRECTLY TO THE LINEARIZATION POINT!
413  function_system.self_exmap(delta);
414  error_new = function_system.weighted_errors(LINPOINT).squaredNorm();
415  error_diff = error - error_new;
416  // feedback
417  if (!prop.quiet) {
418  cout << "LM Iteration " << num_iter << ": (lambda=" << lambda << ") ";
419  if (error_diff > 0.) {
420  cout << "residual: " << error_new << endl;
421  } else {
422  cout << "rejected" << endl;
423  }
424  }
425  // decide if acceptable
426  if (error_diff > 0.) {
427 
428 #ifndef USE_PDL_STOPPING_CRITERIA
429  if (error_diff < prop.epsilon_rel * error) {
430  break;
431  }
432 #endif
433 
434  // Update lambda
435  lambda /= prop.lm_lambda_factor;
436 
437  // Record the error at the newly-accepted estimate.
438  error = error_new;
439 
440  // Relinearize around the newly-accepted estimate.
441  jacobian = function_system.jacobian();
442 
443 #ifdef USE_PDL_STOPPING_CRITERIA
444  r = function_system.weighted_errors(LINPOINT);
445  g = mul_SparseMatrixTrans_Vector(jacobian, r);
446 #endif
447  } else {
448  // reject new estimate
449  lambda *= prop.lm_lambda_factor;
450  // restore previous estimate
451  function_system.estimate_to_linpoint();
452  }
453 
454  // Compute the step for the next iteration.
455  delta = compute_gauss_newton_step(jacobian, &function_system._R, lambda);
456 
457  } // end while
458 
459  if (num_iterations != NULL) {
460  *num_iterations = num_iter;
461  }
462  // Copy current estimate contained in linpoint.
463  function_system.linpoint_to_estimate();
464 }

466 void Optimizer::powells_dog_leg(int* num_iterations, double delta0,
467  int max_iterations, double epsilon1, double epsilon2, double epsilon3) {
468  // Batch optimization
469  int num_iter = 0;
470  // current estimate is used as new linearization point
471  function_system.estimate_to_linpoint();
472 
473  double delta = delta0;
474  SparseSystem jacobian(1, 1);
475  VectorXd f_x;
476  VectorXd grad;
477 
478  bool found = powells_dog_leg_update(epsilon1, epsilon3, jacobian, f_x, grad);
479 
480  double rho_denominator;
481 
482  while ((not found) && (max_iterations == 0 || num_iter < max_iterations)) {
483  num_iter++;
484  cout << "PDL Iteration " << num_iter << " residual: " << f_x.squaredNorm()
485  << endl;
486  // compute alpha
487  double alpha = grad.squaredNorm() / (jacobian * grad).squaredNorm();
488  // steepest descent
489  VectorXd h_sd = -grad;
490  // solve Gauss Newton
491  VectorXd h_gn = compute_gauss_newton_step(jacobian, &function_system._R);
492  // compute dog leg h_dl
493  // x0 = x: remember (and return) linearization point of R
494  function_system.linpoint_to_estimate();
495  VectorXd h_dl = compute_dog_leg(alpha, h_sd, h_gn, delta, rho_denominator);
496  // Evaluate new solution, update estimate and trust region.
497  if (h_dl.norm() <= epsilon2) {
498  found = true;
499  } else {
500  // new estimate
501  // change linearization point directly (original LP saved in estimate)
502  function_system.self_exmap(h_dl);
503  // calculate gain ratio rho
504  VectorXd f_x_new = function_system.weighted_errors(LINPOINT);
505  double rho = (f_x.squaredNorm() - f_x_new.squaredNorm())
506  / (rho_denominator);
507  if (rho > 0) {
508  // accept new estimate
509  cout << "accepted" << endl;
510  f_x = f_x_new;
511  found = powells_dog_leg_update(epsilon1, epsilon3, jacobian, f_x, grad);
512  } else {
513  // reject new estimate, overwrite with last saved one
514  cout << "rejected" << endl;
515  function_system.estimate_to_linpoint();
516  }
517  if (rho > 0.75) {
518  delta = max(delta, 3.0 * h_dl.norm());
519  } else if (rho < 0.25) {
520  delta *= 0.5;
521  found = (delta <= epsilon2);
522  }
523  }
524  }
525  if (num_iterations) {
526  *num_iterations = num_iter;
527  }
528  // Overwrite potentially rejected linearization point with last saved one
529  // (could be identical if it was accepted in the last iteration).
530 
531  function_system.swap_estimates();
532
