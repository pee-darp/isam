89  Pose3d(const Eigen::MatrixXd& m) {
90  if (m.rows()==6 && m.cols()==1) {
91  _t = Point3d(m(0), m(1), m(2));
92  _rot = Rot3d(m(3), m(4), m(5));
93  } else if (m.rows()==4 && m.cols()==4) {
94  // Convert a homogeneous 4x4 transformation matrix to a Pose3.
95  Eigen::Matrix4d wTo = m / m(3,3); // enforce T(3,3)=1
96  Eigen::Vector3d t = wTo.col(3).head(3);
97  Eigen::Matrix3d wRo = wTo.topLeftCorner(3,3);
98  _t = Point3d(t(0), t(1), t(2));
99  _rot = Rot3d(wRo);
100  } else {
101  require(false, "Pose3d constructor called with matrix of wrong dimension");
102  }
