297 SparseSystem Slam::jacobian_numerical_columnwise() {
298  // label starting points of rows, and initialize sparse row vectors with
299  // correct number of entries
300  int num_rows = _dim_measure;
301  DeleteOnReturn rows_ptr(new SparseVector* [num_rows]);
302  SparseVector** rows = rows_ptr._ptr; //[num_rows];
303  int pos = 0;
304  vector<int> factor_offset(get_factors().size());
305  for (list<Factor*>::const_iterator it = get_factors().begin();
306  it!=get_factors().end();
307  it++) {
308  (*it)->_start = pos;
309  int dimtotal = 0;
310  for (vector<Node*>::const_iterator it2 = (*it)->nodes().begin();
311  it2!=(*it)->nodes().end();
312  it2++) {
313  dimtotal += (*it2)->dim();
314  }
315  for (int i=0; i<(*it)->dim(); i++, pos++) {
316  // do not delete, will be pulled into SparseSystem below
317  rows[pos] = new SparseVector(dimtotal);
318  }
319  }
320  // larger than needed, but avoids some book keeping and avoids many (smaller) allocations
321  VectorXd y_plus(num_rows);
322  VectorXd y_minus(num_rows);
323  int col = 0;
324  for (list<Node*>::const_iterator it = get_nodes().begin(); it!=get_nodes().end(); it++) {
325  Node* node = *it;
326  int dim_node = node->dim();
327  VectorXd delta(dim_node);
328  delta.setZero();
329  VectorXd original = node->vector0();
330  for (int c=0; c<dim_node; c++, col++) {
331  // calculate column for +epsilon
332  delta(c) = epsilon;
333  node->self_exmap(delta);
334  int row = 0;
335  for (list<Factor*>::const_iterator it_factor = node->factors().begin();
336  it_factor!=node->factors().end();
337  it_factor++) {
338  Factor* factor = *it_factor;
339  int dim_factor = factor->dim();
340  y_plus.segment(row, dim_factor) = factor->evaluate();
341  row += dim_factor;
342  }
343  node->update0(original);
344  // calculate column for -epsilon
345  delta(c) = - epsilon;
346  node->self_exmap(delta);
347  row = 0;
348  for (list<Factor*>::const_iterator it_factor = node->factors().begin();
349  it_factor!=node->factors().end();
350  it_factor++) {
351  Factor* factor = *it_factor;
352  int dim_factor = factor->dim();
353  y_minus.segment(row, dim_factor) = factor->evaluate();
354  row += dim_factor;
355  }
356  node->update0(original);
357  delta(c) = 0.; // reusing delta
358  // calculate derivative
359  VectorXd diff = (y_plus.head(row) - y_minus.head(row)) / (epsilon + epsilon);
360  // write entries into sparse Jacobian
361  row = 0;
362  int i = 0;
363  for (list<Factor*>::const_iterator it_factor = (*it)->factors().begin();
364  it_factor!=(*it)->factors().end();
365  it_factor++, i++) {
366  for (int r=0; r<(*it_factor)->dim(); r++, row++) {
367  if (diff(row)!=0.) { // omit 0 entries
368  int offset = (*it_factor)->_start;
369  rows[offset+r]->append(col, diff(row)); // faster than SparseVector.set
370  }
371  }
372  }
373  }
374  }
375  VectorXd rhs = weighted_errors(LINPOINT);
376  return SparseSystem(num_rows, _dim_nodes, rows, rhs);
377 }

395 SparseSystem Slam::jacobian_partial(int last_n) {
396  update_starts();
397  // actual assembly of Jacobian
398  int num_rows = _dim_measure;
399  if (last_n > 0) {
400  num_rows = _num_new_rows;
401  }
402  DeleteOnReturn rows_ptr(new SparseVector*[num_rows]);
403  SparseVector** rows = rows_ptr._ptr; //[num_rows];
404 
405  VectorXd rhs(num_rows);
406  int row = 0;
407  const list<Factor*>& factors = get_factors();
408  list<Factor*>::const_iterator it = factors.begin();
409  if (last_n != -1) {
410  // skip all entries except for last_n
411  for (int n = num_factors(); n>last_n; n--, it++);
412  }
413  for (; it!=factors.end(); it++) {
414  Factor* factor = *it;
415  Jacobian jac = factor->jacobian_internal(_prop.force_numerical_jacobian);
416  VectorXd jac_rhs = jac.rhs();
417  for (int r=0; r<jac_rhs.rows(); r++) {
418  rhs(row+r) = jac_rhs(r);
419  // do not delete, will be pulled into SparseSystem below
420  rows[row+r] = new SparseVector(jac.dimtotal());
421  }
422  for (Terms::const_iterator it=jac.terms().begin(); it!=jac.terms().end(); it++) {
423  int offset = it->node()->_start;
424  int nr = it->term().rows();
425  for (int r=0; r<nr; r++) { // 0-entries not omitted
426  rows[row+r]->set(offset, it->term().row(r));
427  }
428  }
429  row += factor->dim();
430  }
431  return SparseSystem(num_rows, _dim_nodes, rows, rhs);
432 }
