55  static Eigen::Matrix3d euler_to_wRo(double yaw, double pitch, double roll) {
56  double c__ = cos(yaw);
57  double _c_ = cos(pitch);
58  double __c = cos(roll);
59  double s__ = sin(yaw);
60  double _s_ = sin(pitch);
61  double __s = sin(roll);
62  double cc_ = c__ * _c_;
63  double cs_ = c__ * _s_;
64  double sc_ = s__ * _c_;
65  double ss_ = s__ * _s_;
66  double c_c = c__ * __c;
67  double c_s = c__ * __s;
68  double s_c = s__ * __c;
69  double s_s = s__ * __s;
70  double _cc = _c_ * __c;
71  double _cs = _c_ * __s;
72  double csc = cs_ * __c;
73  double css = cs_ * __s;
74  double ssc = ss_ * __c;
75  double sss = ss_ * __s;
76  Eigen::Matrix3d wRo;
77  wRo <<
78  cc_ , css-s_c, csc+s_s,
79  sc_ , sss+c_c, ssc-c_s,
80  -_s_ , _cs, _cc;
81  return wRo;
82  }

100  static Eigen::Quaterniond euler_to_quat(double yaw, double pitch, double roll) {
101  double sy = sin(yaw*0.5);
102  double cy = cos(yaw*0.5);
103  double sp = sin(pitch*0.5);
104  double cp = cos(pitch*0.5);
105  double sr = sin(roll*0.5);
106  double cr = cos(roll*0.5);
107  double w = cr*cp*cy + sr*sp*sy;
108  double x = sr*cp*cy - cr*sp*sy;
109  double y = cr*sp*cy + sr*cp*sy;
110  double z = cr*cp*sy - sr*sp*cy;
111  return Eigen::Quaterniond(w,x,y,z);
112  }

114  static void quat_to_euler(Eigen::Quaterniond q, double& yaw, double& pitch, double& roll) {
115  const double q0 = q.w();
116  const double q1 = q.x();
117  const double q2 = q.y();
118  const double q3 = q.z();
119 // roll = atan2(2*(q0*q1+q2*q3), 1-2*(q1*q1+q2*q2));
120  roll = atan2(2.0*(q0*q1+q2*q3), q0*q0-q1*q1-q2*q2+q3*q3); // numerically more stable (thanks to Dehann for pointing this out)
121  pitch = asin(2.0*(q0*q2-q3*q1));
122 // yaw = atan2(2*(q0*q3+q1*q2), 1-2*(q2*q2+q3*q3));
123  yaw = atan2(2.0*(q0*q3+q1*q2), q0*q0+q1*q1-q2*q2-q3*q3);
124  }

126  static Eigen::Quaterniond delta3_to_quat(const Eigen::Vector3d& delta) {
127  double theta = delta.norm();
128  double S;
129  if (theta < 0.0001) { // sqrt4(machine precession)
130  S = 0.5 + theta*theta/48.;
131  } else {
132  S = sin(0.5*theta)/theta;
133  }
134  double C = cos(0.5*theta);
135  return Eigen::Quaterniond(C, S*delta(0), S*delta(1), S*delta(2));
136  }

229  Rot3d exmap(const Eigen::VectorXd& delta) const {
230 #if 1
231  // direct solution by mapping to quaternion (following Grassia98jgt)
232  Rot3d rot(_quat * delta3_to_quat(delta));
233  return rot;
234 #else
235  // cumbersome and slower solution
236  double theta = delta.norm();
237  double a, b, c;
238  if (theta>1e-10) {
239  a = delta(0)/theta;
240  b = delta(1)/theta;
241  c = delta(2)/theta;
242  } else {
243  a = b = c = 0.;
244  }
245  double S = sin(theta);
246  double C = 1-cos(theta);
247  Eigen::Matrix3d m;
248  m <<
249  1+(-a*a-b*b)*C, a*S-b*c*C, b*S+a*c*C,
250  -a*S-b*c*C, 1+(-a*a-c*c)*C, c*S-a*b*C,
251  -b*S+a*c*C, -c*S-a*b*C, 1+(-b*b-c*c)*C;
252  return Rot3d(wRo() * m);
