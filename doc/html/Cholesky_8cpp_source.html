69  tic("Cholesky");
70 
71  reset(); // make sure _L, _rhs, _order are empty
72 
73  cholmod_sparse* At = to_cholmod_transp(Ab);
74  int nrow = At->ncol;
75  int ncol = At->nrow;
76 
77  // Cholesky factorization
78  // cholmod factors AA' instead of A'A - so we need to pass in At!
79  cholmod_factor *L_factor;
80 #if 0
81  // restrict ordering (default is trying several)
82  Common.nmethods = 1;
83  Common.method[0].ordering = CHOLMOD_AMD; //CHOLMOD_NATURAL; //CHOLMOD_COLAMD;
84  Common.postorder = false;
85 #endif
86  if (lambda>0) { // for Levenberg-Marquardt
87  cholmod_sparse* A = cholmod_transpose(At, 1, &Common);
88  // make symmetric matrix (only upper part saved)
89  cholmod_sparse* AtA = cholmod_ssmult(At, A, 1, 1, 1, &Common);
90  // modify diagonal
91  int* AtAp = (int*)AtA->p;
92  // int* AtAi = (int*)AtA->i;
93  double* AtAx = (double*)AtA->x;
94  for (int i=0; i<ncol; i++) {
95  int p = AtAp[i+1]-1;
96  AtAx[p] *= (1+lambda);
97  }
98  L_factor = cholmod_analyze(AtA, &Common);
99  tic("cholmod_factorize");
100  cholmod_factorize(AtA, L_factor, &Common);
101  toc("cholmod_factorize");
102  cholmod_free_sparse(&AtA, &Common);
103  cholmod_free_sparse(&A, &Common);
104  } else {
105  L_factor = cholmod_analyze(At, &Common);
106  tic("cholmod_factorize");
107  cholmod_factorize(At, L_factor, &Common);
108  toc("cholmod_factorize");
109  }
110  // make sure factorization is in correct format (LL, simplicial, packed, ordered)
111  cholmod_change_factor(CHOLMOD_REAL, true, false, true, true, L_factor, &Common);
112 
113  // calculate new rhs by forward substitution (y in R'y = A'b)
114  // note: original rhs is size nrow, Atb and new rhs are size ncol
115  cholmod_dense* A_rhs = cholmod_zeros(nrow, 1, CHOLMOD_REAL, &Common);
116  memcpy(A_rhs->x, Ab.rhs().data(), nrow*sizeof(double));
117  cholmod_dense* Atb = cholmod_zeros(ncol, 1, CHOLMOD_REAL, &Common);
118  double alpha[2] = {1., 0.}; // Atb = 1 * (At*A_rhs)
119  double beta[2] = {0., 0.}; // + 0 * Atb
120  cholmod_sdmult(At, 0, alpha, beta, A_rhs, Atb, &Common);
121  // permute Atb according to L_factor
122  cholmod_dense* Atb_perm = cholmod_solve(CHOLMOD_P, L_factor, Atb, &Common);
123  // forward sub to obtain new (modified) rhs
124  _rhs = cholmod_solve(CHOLMOD_L, L_factor, Atb_perm, &Common);
125 
126  // optionally solve the triangular system by backsubstitution
127  if (delta) {
128  cholmod_dense* delta_ = cholmod_solve(CHOLMOD_Lt, L_factor, _rhs, &Common);
129  *delta = VectorXd(_rhs->nrow);
130  memcpy(delta->data(), (double*)delta_->x, _rhs->nrow*sizeof(double));
131  cholmod_free_dense(&delta_, &Common);
132  }
133 
134  // create R/L with ordering and rhs
135  // WARNING: L_factor becomes symbolic!! (numeric L is literally pulled out for efficiency)
136  _order = new int[ncol];
137  memcpy(_order, (int*)L_factor->Perm, ncol*sizeof(int));
138  _L = cholmod_factor_to_sparse(L_factor, &Common);
139 
140  cholmod_free_dense(&Atb_perm, &Common);
141  cholmod_free_dense(&Atb, &Common);
142  cholmod_free_dense(&A_rhs, &Common);
143  cholmod_free_factor(&L_factor, &Common);
144  cholmod_free_sparse(&At, &Common);
145 
146  toc("Cholesky");

248  int* cholesky(cs* csA, cs* csAt, int n, double lambda, css* S, csn* N) {
249  // Cholesky factorization
250  cs* csAtA = cs_multiply(csAt, csA);
251  if (lambda>0.) {
252  // modify diagonal
253  for (int i=0; i<n; i++) {
254  int p = csAtA->p[i];
255  while (csAtA->i[p]!=i) {p++;} // find diagonal entry
256  csAtA->x[p] *= (1.+lambda);
257  }
258  }
259  S = cs_schol(1, csAtA);
260  N = cs_chol(csAtA, S);
261  // straight copy
262  _L = cs_spalloc(n, n, N->L->nzmax, 1, 0);
263  memcpy(_L->p, N->L->p, (n+1) * sizeof(int));
264  memcpy(_L->i, N->L->i, N->L->nzmax * sizeof(int));
265  memcpy(_L->x, N->L->x, N->L->nzmax * sizeof(double));
266  int* p = cs_pinv(S->pinv, csAtA->n);
267  cs_spfree(csAtA);
268  return p;
269  }

271  void factorize(const SparseSystem& Ab, VectorXd* delta = NULL, double lambda = 0) {
272  tic("Cholesky");
273 
274  reset(); // make sure _L, _rhs, _order are empty
275 
276  // QR with correct rhs
277  cs* csAt = to_csparse_transp(Ab);
278  cs* csA = cs_transpose(csAt, 1);
279  int m = csA->m;
280  int n = csA->n;
281  css* S = NULL;
282  csn* N = NULL;
283  int* p = NULL;
284  if (USE_CSPARSE_QR) {
285  cout << "WARNING: Using slow QR factorization without LM support" << endl;
286  p = qr(csA, n, S, N);
287  } else {
288  p = cholesky(csA, csAt, n, lambda, S, N);
289  }
290  _order = new int[n];
291  memcpy(_order, p, n*sizeof(int));
292  // reorder rhs
293  const double* Ab_rhs = Ab.rhs().data();
294  // result Atb needs to be initialized - gets added!
295  double *Atb = new double[m];
296  for (int i=0; i<n; i++) {
297  Atb[i] = 0.;
298  }
299  cs_gaxpy(csAt, Ab_rhs, Atb);
300 
301  // note that rhs is size n, Atb and B_rhs are longer (size m)
302  _rhs = new double[n];
303  for (int i=0; i<n; i++) {
304  _rhs[i] = Atb[p[i]]; // permute Atb
305  }
306 
307  // forward sub to include R
308  cs_lsolve(_L, _rhs);
309 
310  // optionally solve the triangular system
311  if (delta) {
312  double* delta_ = new double[n];
313  memcpy(delta_, _rhs, n*sizeof(double));
314  cs_ltsolve(_L, delta_);
315  *delta = VectorXd(n);
316  memcpy(delta->data(), delta_, n*sizeof(double));
317  delete[] delta_;
318  }
319 
320  // clean up
321  if (USE_CSPARSE_QR) {
322  cs_free(p);
323  }
324  cs_spfree(csA);
325  cs_spfree(csAt);
326  cs_sfree(S);
327  cs_nfree(N);
328  delete[] Atb;
329  }
